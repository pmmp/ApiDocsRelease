50 public static function getSubChunkCount(Chunk $chunk) : int{
51 for($y = Chunk::MAX_SUBCHUNK_INDEX, $count = count($chunk->getSubChunks()); $y >= Chunk::MIN_SUBCHUNK_INDEX; --$y, --$count){
52 if($chunk->getSubChunk($y)->isEmptyFast()){
53 continue;
54 }
55 return $count;
56 }
57
58 return 0;
59 }

61 public static function serializeFullChunk(Chunk $chunk, BlockTranslator $blockTranslator, PacketSerializerContext $encoderContext, ?string $tiles = null) : string{
62 $stream = PacketSerializer::encoder($encoderContext);
63
64 $subChunkCount = self::getSubChunkCount($chunk);
65 $writtenCount = 0;
66 for($y = Chunk::MIN_SUBCHUNK_INDEX; $writtenCount < $subChunkCount; ++$y, ++$writtenCount){
67 self::serializeSubChunk($chunk->getSubChunk($y), $blockTranslator, $stream, false);
68 }
69
70 $biomeIdMap = LegacyBiomeIdToStringIdMap::getInstance();
71 //all biomes must always be written :(
72 for($y = Chunk::MIN_SUBCHUNK_INDEX; $y <= Chunk::MAX_SUBCHUNK_INDEX; ++$y){
73 self::serializeBiomePalette($chunk->getSubChunk($y)->getBiomeArray(), $biomeIdMap, $stream);
74 }
75
76 $stream->putByte(0); //border block array count
77 //Border block entry format: 1 byte (4 bits X, 4 bits Z). These are however useless since they crash the regular client.
78
79 if($tiles !== null){
80 $stream->put($tiles);
81 }else{
82 $stream->put(self::serializeTiles($chunk));
83 }
84 return $stream->getBuffer();
85 }

87 public static function serializeSubChunk(SubChunk $subChunk, BlockTranslator $blockTranslator, PacketSerializer $stream, bool $persistentBlockStates) : void{
88 $layers = $subChunk->getBlockLayers();
89 $stream->putByte(8); //version
90
91 $stream->putByte(count($layers));
92
93 $blockStateDictionary = $blockTranslator->getBlockStateDictionary();
94
95 foreach($layers as $blocks){
96 $bitsPerBlock = $blocks->getBitsPerBlock();
97 $words = $blocks->getWordArray();
98 $stream->putByte(($bitsPerBlock << 1) | ($persistentBlockStates ? 0 : 1));
99 $stream->put($words);
100 $palette = $blocks->getPalette();
101
102 if($bitsPerBlock !== 0){
103 //these LSHIFT by 1 uvarints are optimizations: the client expects zigzag varints here
104 //but since we know they are always unsigned, we can avoid the extra fcall overhead of
105 //zigzag and just shift directly.
106 $stream->putUnsignedVarInt(count($palette) << 1); //yes, this is intentionally zigzag
107 }
108 if($persistentBlockStates){
109 $nbtSerializer = new NetworkNbtSerializer();
110 foreach($palette as $p){
111 //TODO: introduce a binary cache for this
112 $state = $blockStateDictionary->generateDataFromStateId($blockTranslator->internalIdToNetworkId($p));
113 if($state === null){
114 $state = $blockTranslator->getFallbackStateData();
115 }
116
117 $stream->put($nbtSerializer->write(new TreeRoot($state->toNbt())));
118 }
119 }else{
120 foreach($palette as $p){
121 $stream->put(Binary::writeUnsignedVarInt($blockTranslator->internalIdToNetworkId($p) << 1));
122 }
123 }
124 }
125 }

127 private static function serializeBiomePalette(PalettedBlockArray $biomePalette, LegacyBiomeIdToStringIdMap $biomeIdMap, PacketSerializer $stream) : void{
128 $biomePaletteBitsPerBlock = $biomePalette->getBitsPerBlock();
129 $stream->putByte(($biomePaletteBitsPerBlock << 1) | 1); //the last bit is non-persistence (like for blocks), though it has no effect on biomes since they always use integer IDs
130 $stream->put($biomePalette->getWordArray());
131
132 //these LSHIFT by 1 uvarints are optimizations: the client expects zigzag varints here
133 //but since we know they are always unsigned, we can avoid the extra fcall overhead of
134 //zigzag and just shift directly.
135 $biomePaletteArray = $biomePalette->getPalette();
136 if($biomePaletteBitsPerBlock !== 0){
137 $stream->putUnsignedVarInt(count($biomePaletteArray) << 1);
138 }
139
140 foreach($biomePaletteArray as $p){
141 if($biomeIdMap->legacyToString($p) === null){
142 //make sure we aren't sending bogus biomes - the 1.18.0 client crashes if we do this
143 $p = BiomeIds::OCEAN;
144 }
145 $stream->put(Binary::writeUnsignedVarInt($p << 1));
146 }
147 }

149 public static function serializeTiles(Chunk $chunk) : string{
150 $stream = new BinaryStream();
151 foreach($chunk->getTiles() as $tile){
152 if($tile instanceof Spawnable){
153 $stream->put($tile->getSerializedSpawnCompound()->getEncodedNbt());
154 }
155 }
156
157 return $stream->getBuffer();
158 }
