56 public function missingOrWrongTypeException(string $name, ?Tag $tag) : BlockStateDeserializeException{
57 return new BlockStateDeserializeException("Property \"$name\" " . ($tag !== null ? "has unexpected type " . get_class($tag) : "is missing"));
58 }
59
60 public function badValueException(string $name, string $stringifiedValue, ?string $reason = null) : BlockStateDeserializeException{
61 return new BlockStateDeserializeException(
62 "Property \"$name\" has unexpected value \"$stringifiedValue\"" . (
63 $reason !== null ? " ($reason)" : ""
64 ));
65 }

68 public function readBool(string $name) : bool{
69 $this->usedStates[$name] = true;
70 $tag = $this->data->getState($name);
71 if($tag instanceof ByteTag){
72 switch($tag->getValue()){
73 case 0: return false;
74 case 1: return true;
75 default: throw $this->badValueException($name, (string) $tag->getValue());
76 }
77 }
78 throw $this->missingOrWrongTypeException($name, $tag);
79 }
80
82 public function readInt(string $name) : int{
83 $this->usedStates[$name] = true;
84 $tag = $this->data->getState($name);
85 if($tag instanceof IntTag){
86 return $tag->getValue();
87 }
88 throw $this->missingOrWrongTypeException($name, $tag);
89 }

92 public function readBoundedInt(string $name, int $min, int $max) : int{
93 $result = $this->readInt($name);
94 if($result < $min || $result > $max){
95 throw $this->badValueException($name, (string) $result, "Must be inside the range $min ... $max");
96 }
97 return $result;
98 }

101 public function readString(string $name) : string{
102 $this->usedStates[$name] = true;
103 //TODO: only allow a specific set of values (strings are primarily used for enums)
104 $tag = $this->data->getState($name);
105 if($tag instanceof StringTag){
106 return $tag->getValue();
107 }
108 throw $this->missingOrWrongTypeException($name, $tag);
109 }

117 private function parseFacingValue(int $value, array $mapping) : int{
118 $result = $mapping[$value] ?? null;
119 if($result === null){
120 throw new BlockStateDeserializeException("Unmapped facing value " . $value);
121 }
122 return $result;
123 }

126 public function readFacingDirection() : int{
127 return $this->parseFacingValue($this->readInt(BlockStateNames::FACING_DIRECTION), [
128 0 => Facing::DOWN,
129 1 => Facing::UP,
130 2 => Facing::NORTH,
131 3 => Facing::SOUTH,
132 4 => Facing::WEST,
133 5 => Facing::EAST
134 ]);
135 }

138 public function readBlockFace() : int{
139 return match($raw = $this->readString(BlockStateNames::MC_BLOCK_FACE)){
140 StringValues::MC_BLOCK_FACE_DOWN => Facing::DOWN,
141 StringValues::MC_BLOCK_FACE_UP => Facing::UP,
142 StringValues::MC_BLOCK_FACE_NORTH => Facing::NORTH,
143 StringValues::MC_BLOCK_FACE_SOUTH => Facing::SOUTH,
144 StringValues::MC_BLOCK_FACE_WEST => Facing::WEST,
145 StringValues::MC_BLOCK_FACE_EAST => Facing::EAST,
146 default => throw $this->badValueException(BlockStateNames::MC_BLOCK_FACE, $raw)
147 };
148 }

154 public function readFacingFlags() : array{
155 $result = [];
156 $flags = $this->readBoundedInt(BlockStateNames::MULTI_FACE_DIRECTION_BITS, 0, 63);
157 foreach([
158 BlockLegacyMetadata::MULTI_FACE_DIRECTION_FLAG_DOWN => Facing::DOWN,
159 BlockLegacyMetadata::MULTI_FACE_DIRECTION_FLAG_UP => Facing::UP,
160 BlockLegacyMetadata::MULTI_FACE_DIRECTION_FLAG_NORTH => Facing::NORTH,
161 BlockLegacyMetadata::MULTI_FACE_DIRECTION_FLAG_SOUTH => Facing::SOUTH,
162 BlockLegacyMetadata::MULTI_FACE_DIRECTION_FLAG_WEST => Facing::WEST,
163 BlockLegacyMetadata::MULTI_FACE_DIRECTION_FLAG_EAST => Facing::EAST
164 ] as $flag => $facing){
165 if(($flags & $flag) !== 0){
166 $result[$facing] = $facing;
167 }
168 }
169
170 return $result;
171 }

174 public function readEndRodFacingDirection() : int{
